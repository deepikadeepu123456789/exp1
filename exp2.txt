import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
cars_df=pd.read_csv('/content/mtcars.csv')
print(cars_df)

a=plt.scatter(cars_df['mpg'],cars_df['wt'],marker='*')
print(a)
plt.xlabel("mpg")
plt.ylabel("weight")
plt.title("relationship between the weight of the car and mpg")
plt.show()

trans=cars_df['am'].value_counts()
print(trans)
trans.plot.bar()
plt.xlabel("transmission")
plt.title('bar plot to check the frequency distribution of transmission type of cars')
plt.show()

cars_df['MPG.city'].plot(kind='box') # This column does not exist in mtcars.csv
plt.xlabel('mpg')
plt.title('box and whisker plot of mpg')
plt.show()

min_value=np.min(cars_df['Price']) # This column does not exist in mtcars.csv
q1=np.percentile(cars_df['Price'],25)
median=np.median(cars_df['Price'])
q3=np.percentile(cars_df['Price'],75)
max_value=np.max(cars_df['Price'])
print('minimum value',min_value)
print('first quartile',q1)
print('median',median)
print('third quartile',q3)
print('maximum value',max_value)
